[cluster LSDYNA]
FormLayout = selectionpanel
Category = Applications
IconUrl = https://raw.githubusercontent.com/hirtanak/scripts/master/LS-DYNAlogo.png

Autoscale = $Autoscale

    [[node defaults]]
    UsePublicNetwork = $UsePublicNetwork
    Credentials = $Credentials    
    SubnetId = $SubnetId
    Region = $Region
    KeyPairLocation = ~/.ssh/cyclecloud.pem
#    KeyPairLocation = ~/examples/id_rsaccuser 
   
        [[[cluster-init LSDYNA:default:1.0.1]]]
        Optional = True

        [[[configuration]]]
        LSDYNA_VERSION = $configuration_LSDYNA_version
        LICENSE = $configuration_LICENSE
        MPI = $configuration_MPI
	MODEL = $configuration_Model

    [[node master]]
    MachineType = $MasterMachineType
    IsReturnProxy = $ReturnProxy
    AdditionalClusterInitSpecs = $MasterClusterInitSpecs
    ImageName = $ImageName

        [[[configuration]]]

        [[[cluster-init LSDYNA:master:1.0.1]]]
        Optional = True

        [[[network-interface eth0]]]
        AssociatePublicIpAddress = $UsePublicNetwork
	StaticPublicIpAddress = true

        [[[input-endpoint ganglia]]]
        PrivatePort = 8652
        PublicPort = 8652

        # Add 2 premium disks in a RAID 0 configuration to the NFS export
        [[[volume nfs-1]]]
        Size = 512
        SSD = True
        Mount = nfs
        Persistent = true

        [[[volume nfs-2]]]
        Size = 512
        SSD = True
        Mount = nfs
        Persistent = true

	[[[configuration cyclecloud.mounts.nfs]]]
	mountpoint = /mnt/exports/apps
	fs_type = ext4
	raid_level = 0

        [[[configuration cyclecloud.exports.nfs_data]]]
        type = nfs
        export_path = /mnt/exports/apps

    [[nodearray execute]]
    MachineType = $ExecuteMachineType
    MaxCoreCount = $MaxExecuteCoreCount
    Interruptible = $UseLowPrio
    AdditionalClusterInitSpecs = $ExecuteClusterInitSpecs
    InstallJetpack = true
    JetpackPlatform = centos-7
    ImageName = $ImageNameExecute
    Azure.SingleScaleset = true

        [[[configuration]]]

        [[[cluster-init LSDYNA:execute:1.0.1]]]
        Optional = True

        [[[network-interface eth0]]]
        AssociatePublicIpAddress = $ExecuteNodesPublic

        [[[configuration cyclecloud.mounts.nfs_data]]]
        type = nfs
        mountpoint = /mnt/exports/apps
        export_path = /mnt/exports/apps

            Label = Standard_HC44rs

    [[parameters Auto-Scaling]]
    Description = "The cluster can autoscale to the workload, adding execute hosts as jobs are queued. To enable this check the box below and choose the initial and maximum core counts for the cluster"

        [[[parameter Autoscale]]]
        Label = Autoscale
        DefaultValue = true
        Widget.Plugin = pico.form.BooleanCheckBox
        Widget.Label = Start and stop execute instances automatically

        [[[parameter MaxExecuteCoreCount]]]
        Label = Max Cores

        Config.IntegerOnly = true

        [[[parameter UseLowPrio]]]
        Label = Low Priority
        DefaultValue = true
        Widget.Plugin = pico.form.BooleanCheckBox
        Widget.Label = Use low priority instances for execute hosts

    [[parameters Networking]]
    Order = 14

        [[[parameter SubnetId]]]

        ParameterType = Azure.Subnet
        Required = True
